  SOURCE_DIR "${CMAKE_SOURCE_DIR}/dep/libmpq/"

  CONFIGURE_COMMAND "${CMAKE_SOURCE_DIR}/dep/libmpq/autogen.sh"
  BUILD_COMMAND "${CMAKE_SOURCE_DIR}/dep/libmpq/configure"
  INSTALL_COMMAND ""
)
